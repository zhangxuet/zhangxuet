    获取url？后边参数

    function translateParam (url) {
    let obj = {}
    let index = url.indexOf('?')
    let str = url.substr(index+1)
    let newArr = str.split('&')
    newArr.forEach(item => {
        let arr = item.split('=')
        obj[arr[0]] = arr[1]
    })
    return obj
}
let str = "http://www.baidu.com/a?name='zs'&age='18'&gender='man'"
let ret = translateParam(str)
console.log(ret)

function translateParam () {
    let obj = {}
    let newArr = []
    newArr = location.href.split('?')[1].split('&').forEach(item => {
        let arr = item.split('=')
        obj[arr[0]] = arr[1]
    })
    return obj
}
